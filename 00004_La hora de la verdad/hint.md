def mitad(un_numero):

def suma_longitudes(un_string, otro_string): 
  return len(un_string) + len(otro_string)  
  
def es_pregunta(oracion):
  return str.startswith(oracion, "¿") and str.endswith(oracion, "?") 
  
def es_mas_largo_que(un_string, otro_string):
  return len(un_string) > len(otro_string)
